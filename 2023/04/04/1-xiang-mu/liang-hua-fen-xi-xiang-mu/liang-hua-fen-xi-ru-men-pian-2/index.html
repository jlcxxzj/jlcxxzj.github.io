运行源代码

import  sys
#被引模块路径
sys.path.append("F:/测试系统/量化交易/freqtrade-develop/freqtrade-develop/freqtrade/")
#编辑器当前打开根文件夹路径
sys.path.append("F:/测试系统/量化交易/freqtrade-develop/freqtrade-develop/")
from  freqtrade.util.gc_setup  import  gc_set_threshold
